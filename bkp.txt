import { json } from "@remix-run/node";
import { webhookHandler } from "../shopify.server.js"; // ‚úÖ centralized webhook validation
import { forwardToWebhookSite } from "../utils/forwardToWebhookSite.js";

export async function action({ request }) {
  console.log("üì• Webhook request received: orders/create");

  const topic = request.headers.get("x-shopify-topic");
  const shop = request.headers.get("x-shopify-shop-domain");

  // ‚úÖ read body once as text
  const rawBody = await request.text();
  let payload;

  try {
    // ‚úÖ Pass rawBody to webhookHandler (so it can validate HMAC)
    const response = await webhookHandler(rawBody, request.headers);
    if (!response.ok) {
      console.warn("‚ö†Ô∏è HMAC validation skipped (likely dev/local test)");
    }
    payload = JSON.parse(rawBody);
  } catch (err) {
    console.warn("‚ö†Ô∏è HMAC validation failed, using fallback:", err.message);
    payload = JSON.parse(rawBody); // ‚úÖ safe, already read
  }

  const orderId = payload?.id;
  if (!orderId) {
    console.warn("‚ö†Ô∏è Missing order ID in payload");
    return json({ error: "Invalid payload" }, { status: 400 });
  }

  console.log(`‚úÖ Order created: ${orderId} from shop ${shop}`);

  const responseObj = json({ success: true });

  forwardToWebhookSite({
    url: `${process.env.SHOPIFY_NEXT_URI}/api/shopify/orders`,
    topic,
    shop,
    payload,
  }).catch((err) => console.error("‚ùå Forwarding failed:", err));

  return responseObj;
}
